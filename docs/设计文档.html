时间管控器

设计文档 V1.0
创建日期: 2026-05-22

项目概述

一个基于ESP32的时间管控设备，用于控制家庭中儿童使用Switch游戏机的时间。通过NFC检测判断游戏机是否在盒内，超出设定时间未检测到时触发声光报警。

项目 描述
项目名称 Switch Time Controller
硬件平台 ESP32 + PN532 NFC模块
开发语言 Arduino C (PlatformIO)
核心功能 NFC检测 + 超时报警 + WiFi配置

⚠️ 重要：NFC标签注册机制

为什么需要注册？

如果随便一个NFC标签放在盒子里就识别为"Switch在盒内"，那么计时永远不超时。

必须先将Switch上的NFC标签UID添加到白名单，只有白名单内的UID才会被识别。

注册方式：通过WiFi连接设备Web管理页面进行"学习"注册

核心流程（改进版）

void loop() {
    // 1. 检查配网按钮（长按3秒）
    if (buttonManager.isLongPressed(3000)) {
        enterConfigMode();  // 进入配网
    }

    // 2. 检查NFC标签
    String detectedUID = nfcManager.detectTag();
    bool isRegistered = nfcManager.isTagRegistered(detectedUID);

    switch (currentState) {
        case IDLE:
            if (isRegistered) {
                currentState = ACTIVE;
                timerManager.reset();
                ledController.setGreen();
                buzzerController.stop();
            }
            break;

        case ACTIVE:
            if (!isRegistered) {
                currentState = COUNTING;
                timerManager.startCounting();
                ledController.setGreenBlink();
            }
            break;

        case COUNTING:
            if (isRegistered) {
                currentState = ACTIVE;
                timerManager.reset();
                ledController.setGreen();
                buzzerController.stop();
            } else if (timerManager.isTimeout()) {
                currentState = WARNING;
                ledController.setRedBlink();
                buzzerController.startAlarm();
                notifyServer.sendTimeoutAlert();
            }
            break;

        case WARNING:
            if (isRegistered) {
                currentState = IDLE;
                timerManager.reset();
                ledController.setBlueBlink();  // 或其他恢复提示
                buzzerController.stop();
            }
            break;
    }

    // 3. 处理Web请求
    webServer.handleClient();

    delay(50);
}

UID白名单验证逻辑

// NfcManager 伪代码
class NfcManager {
    String getCurrentTagUID();          // 读取当前NFC标签UID
    bool isTagRegistered(String uid);   // 检查UID是否在白名单中
    void registerTag(String uid);       // 添加UID到白名单
    void unregisterTag(String uid);     // 从白名单删除UID

    // 主检测函数
    bool detectAndValidate() {
        String uid = getCurrentTagUID();
        if (uid.isEmpty()) return false;
        return isTagRegistered(uid);   // 必须白名单验证通过
    }
};
